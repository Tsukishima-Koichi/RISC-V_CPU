# ==========================================
# 第一部分：U型指令 (Upper Immediate) - 2条
# ==========================================
0x00: lui   x1, 0x12345       # 1. LUI: x1 = 0x12345000
0x04: auipc x2, 0             # 2. AUIPC: x2 = PC + 0 (这里PC=4，所以x2=4)

# ==========================================
# 第二部分：I型算术逻辑指令 - 9条
# ==========================================
0x08: addi  x3, x0, 15        # 3. ADDI: x3 = 15
0x0C: slti  x4, x3, 20        # 4. SLTI: x4 = (15 < 20) ? 1 : 0
0x10: sltiu x5, x3, 20        # 5. SLTIU: 无符号比较
0x14: xori  x6, x3, 15        # 6. XORI: x6 = 15 ^ 15 = 0
0x18: ori   x7, x3, 15        # 7. ORI: x7 = 15 | 15 = 15
0x1C: andi  x8, x3, 15        # 8. ANDI: x8 = 15 & 15 = 15
0x20: slli  x9, x3, 2         # 9. SLLI: x9 = 15 << 2 = 60
0x24: srli  x10, x3, 2        # 10. SRLI: 逻辑右移
0x28: srai  x11, x3, 2        # 11. SRAI: 算术右移

# ==========================================
# 第三部分：R型算术逻辑指令 - 10条
# ==========================================
0x2C: add   x12, x3, x3       # 12. ADD: 15 + 15 = 30
0x30: sub   x13, x3, x3       # 13. SUB: 15 - 15 = 0
0x34: sll   x14, x3, x3       # 14. SLL: 逻辑左移
0x38: slt   x15, x3, x3       # 15. SLT: 有符号比较 (15<15)=0
0x3C: sltu  x16, x3, x3       # 16. SLTU: 无符号比较
0x40: xor   x17, x3, x3       # 17. XOR: 15 ^ 15 = 0
0x44: srl   x18, x3, x3       # 18. SRL: 逻辑右移
0x48: sra   x19, x3, x3       # 19. SRA: 算术右移
0x4C: or    x20, x3, x3       # 20. OR
0x50: and   x21, x3, x3       # 21. AND

# ==========================================
# 第四部分：访存指令 Load/Store - 8条
# ==========================================
0x54: lui   x6, 0x80100       # 准备基址: x6 = 0x80100000 (你的DRAM起始地址)
0x58: sw    x3, 0(x6)         # 22. SW: 存入字
0x5C: sh    x3, 4(x6)         # 23. SH: 存入半字
0x60: sb    x3, 6(x6)         # 24. SB: 存入字节
0x64: lw    x22, 0(x6)        # 25. LW: 读出字
0x68: lh    x23, 4(x6)        # 26. LH: 读出半字(有符号扩展)
0x6C: lhu   x24, 4(x6)        # 27. LHU: 读出半字(无符号扩展)
0x70: lb    x25, 6(x6)        # 28. LB: 读出字节(有符号扩展)
0x74: lbu   x26, 6(x6)        # 29. LBU: 读出字节(无符号扩展)

# ==========================================
# 第五部分：B型分支跳转指令 - 6条
# ==========================================
# 这里采用的逻辑是：如果分支成立，就向前跳 8 个字节（跳过紧跟的 NOP）
0x78: beq   x3, x3, +8        # 30. BEQ: 相等则跳过 NOP
0x7C: nop                     # (被跳过)
0x80: bne   x3, x0, +8        # 31. BNE: 不等则跳过 NOP
0x84: nop                     # (被跳过)
0x88: blt   x0, x3, +8        # 32. BLT: 0 < 15, 跳过 NOP
0x8C: nop                     # (被跳过)
0x90: bge   x3, x0, +8        # 33. BGE: 15 >= 0, 跳过 NOP
0x94: nop                     # (被跳过)
0x98: bltu  x0, x3, +8        # 34. BLTU: 无符号跳过
0x9C: nop                     # (被跳过)
0xA0: bgeu  x3, x0, +8        # 35. BGEU: 无符号跳过
0xA4: nop                     # (被跳过)

# ==========================================
# 第六部分：J型无条件跳转 - 2条
# ==========================================
0xA8: jal   x27, +8           # 36. JAL: 无条件跳转+8，同时x27记录返回地址(0xAC)
0xAC: nop                     # (被跳过)
0xB0: jalr  x28, x27, 0       # 37. JALR: 跳转到 x27 记录的地址 (跳回0xAC)
                              # 注意：由于流水线测试，跳回 0xAC 后会再往下执行一次 JALR

# 结束
0xB4: jal   x0, 0             # 死循环 (Halt) PC = 0xB4
